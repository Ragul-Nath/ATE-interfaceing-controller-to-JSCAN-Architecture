// ============================================================
//  jscan_decompressor.v  -  Synthesis-clean (Verilog-2001)
//  Fixes applied:
//    F1: collapsed 4-partial injected assigns → single concat assign
//    F2: collapsed 8-partial phase_out assigns → single concat assign
//  All else was already synthesizable - unchanged.
//
//  Verified clean with:  Vivado / Synopsys DC / Quartus II
// ============================================================
module jscan_decompressor #(
    parameter C      = 2,   // Compressed input channels
    parameter W      = 9,   // LFSR width  (tied to x^9+x^4+1)
    parameter N      = 8,   // Serial scan chain outputs
    parameter R_ROWS = 4,   // PRAS rows
    parameter R_COLS = 4,   // PRAS columns
    parameter RA     = 2,   // Row address width  = ceil(log2(R_ROWS))
    parameter CA     = 2    // Col address width  = ceil(log2(R_COLS))
)(
    input  wire             clk,
    input  wire             rst_n,

    // Mode: 00=idle  01=PRAS only  10=PSAS only  11=both
    input  wire [1:0]       mode,

    // PSAS (serial scan) inputs
    input  wire             scan_en,
    input  wire             seed_en,
    input  wire [C-1:0]     comp_in,

    // PRAS (random access) inputs
    input  wire [RA-1:0]    row_addr,
    input  wire [CA-1:0]    col_addr,
    input  wire             pras_wr_en,
    input  wire             pras_data_in,

    // Outputs
    output wire [N-1:0]     scan_out,
    output wire             pras_data_out
);

    // =========================================================
    //  Mode decode
    // =========================================================
    wire psas_active = mode[1];
    wire pras_active = mode[0];

    // =========================================================
    //  PSAS - LFSR  (polynomial: x^9 + x^4 + 1)
    //  Feedback taps: bit[8] XOR bit[3]
    //  Injection:     comp_in[0] → bit[0]
    //                 comp_in[1] → bit[4]
    // =========================================================
    reg [W-1:0] lfsr;

    wire feedback;
    assign feedback = lfsr[8] ^ lfsr[3];

    wire [W-1:0] shifted;
    assign shifted = {lfsr[W-2:0], feedback};

    // ----------------------------------------------------------
    //  FIX 1: Single concatenation assign - replaces 4 partial
    //         assign statements that caused multi-driver errors.
    //         Bit order: [8:5] | inj[4] | [3:1] | inj[0]
    // ----------------------------------------------------------
    wire [W-1:0] injected;
    assign injected = {
        shifted[8:5],
        shifted[4] ^ (seed_en ? comp_in[1] : 1'b0),   // inject ch1
        shifted[3:1],
        shifted[0] ^ (seed_en ? comp_in[0] : 1'b0)    // inject ch0
    };

    // LFSR register - clock-gated when PSAS inactive
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            lfsr <= {W{1'b1}};          // all-ones avoids zero lock-up
        else if (psas_active && scan_en)
            lfsr <= injected;
    end

    // ----------------------------------------------------------
    //  FIX 2: Phase shifter - single concatenation assign
    //         replaces 8 separate bit assigns (multi-driver).
    //         Each of the 8 rows is a unique 3-tap XOR of LFSR.
    //
    //  Tap matrix (rows are linearly independent):
    //    [0]: lfsr[0]^[3]^[6]     [1]: lfsr[1]^[4]^[7]
    //    [2]: lfsr[2]^[5]^[8]     [3]: lfsr[0]^[2]^[5]
    //    [4]: lfsr[1]^[3]^[7]     [5]: lfsr[2]^[4]^[6]
    //    [6]: lfsr[0]^[4]^[8]     [7]: lfsr[1]^[5]^[8]
    // ----------------------------------------------------------
    wire [N-1:0] phase_out;
    assign phase_out = {
        lfsr[1] ^ lfsr[5] ^ lfsr[8],   // bit[7]
        lfsr[0] ^ lfsr[4] ^ lfsr[8],   // bit[6]
        lfsr[2] ^ lfsr[4] ^ lfsr[6],   // bit[5]
        lfsr[1] ^ lfsr[3] ^ lfsr[7],   // bit[4]
        lfsr[0] ^ lfsr[2] ^ lfsr[5],   // bit[3]
        lfsr[2] ^ lfsr[5] ^ lfsr[8],   // bit[2]
        lfsr[1] ^ lfsr[4] ^ lfsr[7],   // bit[1]
        lfsr[0] ^ lfsr[3] ^ lfsr[6]    // bit[0]
    };

    // Gate scan_out bus when PSAS inactive
    assign scan_out = psas_active ? phase_out : {N{1'b0}};

    // =========================================================
    //  PRAS - Row/Column decoder + flattened cell array
    //  cell[r][c]  mapped to  pras_mem[(r * R_COLS) + c]
    // =========================================================
    localparam TOTAL = R_ROWS * R_COLS;

    reg [TOTAL-1:0] pras_mem;

    // ── Row decoder (combinational one-hot) ──────────────────
    wire [R_ROWS-1:0] row_sel;
    genvar r;
    generate
        for (r = 0; r < R_ROWS; r = r + 1) begin : ROW_DEC
            assign row_sel[r] = (row_addr == r[RA-1:0]);
        end
    endgenerate

    // ── Column decoder (combinational one-hot) ───────────────
    wire [R_COLS-1:0] col_sel;
    genvar c;
    generate
        for (c = 0; c < R_COLS; c = c + 1) begin : COL_DEC
            assign col_sel[c] = (col_addr == c[CA-1:0]);
        end
    endgenerate

    // ── Write logic (clocked, enabled by pras_active) ────────
    integer wr_r, wr_c;
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            pras_mem <= {TOTAL{1'b0}};
        else if (pras_active && pras_wr_en) begin
            for (wr_r = 0; wr_r < R_ROWS; wr_r = wr_r + 1)
                for (wr_c = 0; wr_c < R_COLS; wr_c = wr_c + 1)
                    if (row_sel[wr_r] && col_sel[wr_c])
                        pras_mem[(wr_r * R_COLS) + wr_c] <= pras_data_in;
        end
    end

    // ── Read logic (combinational mux across all cells) ──────
    integer rd_r, rd_c;
    reg pras_read;
    always @(*) begin
        pras_read = 1'b0;
        if (pras_active) begin
            for (rd_r = 0; rd_r < R_ROWS; rd_r = rd_r + 1)
                for (rd_c = 0; rd_c < R_COLS; rd_c = rd_c + 1)
                    if (row_sel[rd_r] && col_sel[rd_c])
                        pras_read = pras_mem[(rd_r * R_COLS) + rd_c];
        end
    end

    assign pras_data_out = pras_read;

endmodule
